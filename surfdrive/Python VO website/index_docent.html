def zing_couplet( onderdeel, actie ):
    print( "De " + onderdeel + " van de bus " + actie )
    print( actie +", " + actie )
    print( "De " + onderdeel + " van de bus " + actie )
    print( "Als de bus gaat rijden" )
    print("\n")

zing_couplet("wielen", "draaien rond")
zing_couplet("deuren", "gaan open en dicht")
zing_couplet("wissers", "gaan heen en weer")
zing_couplet("lichten", "gaan aan en uit")
