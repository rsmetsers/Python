Opgave Vierkantjes herhalen met een `while`

import turtle


# tekent een vierkant door 4 keer vooruit de gegeven lengte vooruit te lopen en 90 graden te draaien.
def tekenVierkant( lengte ):
    turtle.forward(lengte)
    turtle.right(90)
    turtle.forward(lengte)
    turtle.right(90)
    turtle.forward(lengte)
    turtle.right(90)
    turtle.forward(lengte)
    turtle.right(90)

lengte = 100 # begin waarde
turtle.pendown() # beginnen met tekenen

# teken 10 keer een steeds kleiner wordend vierkant in een cirkel.
while lengte > 0:
    tekenVierkant( lengte )
    turtle.right(15)
    lengte -= 10

turtle.done()

Opgave Vierkantjes herhalen met een `while`


import turtle
#tekent een vierkant door 4 keer vooruit de gegeven lengte vooruit te lopen en 90 graden te draaien.
def tekenVierkant( lengte ):
    aantal_zijdes_getekend = 0
    while aantal_zijdes_getekend < 4: #herhaal 4 keer
        turtle.forward(lengte)
        turtle.right(90)
        aantal_zijdes_getekend += 1

turtle.pendown()
tekenVierkant( 100 )
turtle.done()
