import turtle

def tekenHuis():
    #teken zwarte vierkant
    turtle.forward(150)
    turtle.right(90)
    turtle.forward(150)
    turtle.right(90)
    turtle.forward(150)
    turtle.right(90)
    turtle.forward(150)

    #teken een rood gevulde driehoek
    turtle.fillcolor("red")
    turtle.begin_fill()
    # teken driehoek
    turtle.right(30)
    turtle.forward(150)
    turtle.right(120)
    turtle.forward(150)
    turtle.end_fill()

#teken eerste huis
turtle.pendown()
tekenHuis()
turtle.penup()

#teken tweede huis
turtle.goto(300,0)
turtle.setheading(0)
turtle.pendown()
tekenHuis()
turtle.penup()

#teken derde huis
turtle.goto(-300,0)
turtle.setheading(0)
turtle.pendown()
tekenHuis()
turtle.penup()


turtle.done()

Opgave Nederlandse vlag met functies

Van de code voor het tekenen van een balk maak je een definitie: 
def balk():
    turtle.forward(150)
    turtle.right(90)
    turtle.forward(40)
    turtle.right(90)
    turtle.forward(150)
    turtle.right(90)
    turtle.forward(40)
    turtle.right(90)
Die kun je dan zo aanroepen: `balk()`.
De uiteindelijke code komt er dan zo uit te zien: 
import turtle

def balk():
    turtle.begin_fill()
    turtle.forward(300)
    turtle.right(90)
    turtle.forward(40)
    turtle.right(90)
    turtle.forward(300)
    turtle.right(90)
    turtle.forward(40)
    turtle.right(90)
    turtle.end_fill()

#Ga naar de beginpositie
#turtle.penup()
#turtle.goto(0, 0)

turtle.pendown()
#kleur: rood
turtle.color("red")
#teken eerste balk
balk()
turtle.goto(0, -40)
turtle.color("white")
balk()
turtle.goto(0, -80)
turtle.color("blue")
balk()

# Ga weer naar de beginpositie voor het tekenen van de omlijning
turtle.penup()
turtle.goto(0, 0)
turtle.pendown()

# kleur: zwart
turtle.color("black")
# Teken de zwarte omlijning

Decompositie toepassen

Je hebt in een vorige opdracht een functie geschreven om een huis te tekenen: tekenHuis(). Als je het huis verder opdeelt in logische delen zie je dat het huis uit een vierkant en een driehoek bestaat. Dit heet "decompositie".

Omdat je nu ook al geleerd hebt om met parameters te werken kun je nu logische componenten bouwen.

Opgave Huis tekenen met parameters

import turtle

#FUNCTIE DEFINITIES
#Definitie: teken zwarte vierkant
def tekenVierkant( lengte ):
    turtle.forward(lengte)
    turtle.right(90)
    turtle.forward(lengte)
    turtle.right(90)
    turtle.forward(lengte)
    turtle.right(90)
    turtle.forward(lengte)

#Definitie: teken een rood gevulde driehoek
def tekenDak( lengte, kleur ):
    turtle.fillcolor( kleur )
    turtle.begin_fill()
    # teken driehoek
    turtle.right(30)
    turtle.forward(lengte)
    turtle.right(120)
    turtle.forward(lengte)
    turtle.end_fill()


#HOOFDPROGRAMMA MET AANROEP VAN FUNCTIE
turtle.pendown()
tekenVierkant(150) #Aanroep: tekenVierkant met lengte 150
tekenDak ( 150, "blue") #Aanroep: teken een blauw gevulde driehoek van lengte 150

turtle.done()
