Opgave 12.5.4 Voorkomens tellen

aantalVoorkomensLettersLijst = 26*[0] #lijst voor het bijhouden van namen
print(aantalVoorkomensLettersLijst)

invoer = input("Typ een woord in: ")
woord = invoer.lower() #zet alle letters om in kleine letters

for letter in woord:
    if letter == "a":
        aantalVoorkomensLettersLijst[0] +=1
        print("a gevonden")
    if letter == "b":
        aantalVoorkomensLettersLijst[1] +=1
        print("b gevonden")
    if letter == "c":
        aantalVoorkomensLettersLijst[2] +=1
        print("c gevonden")
    if letter == "d":
        aantalVoorkomensLettersLijst[3] +=1
        print("d gevonden")
    if letter == "e":
        aantalVoorkomensLettersLijst[4] +=1
        print("e gevonden")
    # Hier evt. voor alle andere letters ook afhandelen

print("Dit is een lijst met voorkomens:")
print(aantalVoorkomensLettersLijst)

#de staafdiagram functie:
def staafdiagramAfdrukken(waardes):

    #doorloop alle waardes uit de lijst:
    for waarde in waardes:
        #en print voor elke waarde het bijbehordende aantal *
        print ("*" * waarde)
