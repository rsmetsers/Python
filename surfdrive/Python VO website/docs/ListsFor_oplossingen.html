Afsluitende opgave 12.3 Kaarten schudden

import random

stok = []
for value in ("Aas", "2", "3", "4", "5", "6", "7", "8",
    "10", "Boer", "Vrouw", "Heer"):
    for kleur in ("Harten", "Schoppen", "Klaveren", "Ruiten"):
        stok += [kleur + ' ' + value ]

for i in range( len( stok ) ): #wissel elk kaart met een willekeurige andere
    j = random.randint( i, len( stok )-1 ) #kies een willekeurig getal
    stok[i], stok[j] = stok[j], stok[i]

for kaart in stok:
    print( kaart )
